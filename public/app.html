# Generate Content
curl -X POST ${APP_URL || 'https://your-app.railway.app'}/api/generate \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"prompt":"Write a blog post about AI","toolName":"Blog Writer","tone":"Professional"}'

  -d '{"messages":[{"role":"user","content":"Write me a tagline"}]}'
